Prepare for Hyper-V

# Get the current state of Hyper-V and Windows Subsystem for Linux.
$hyperVEnabled = Get-WmiObject `
    -Class Win32_ComputerSystem `
    -Property HypervisorPresent

$wslEnabled = Get-WindowsOptionalFeature `
    -FeatureName Microsoft-Windows-Subsystem-Linux

# Enable Hyper-V if it is not already enabled.
if (!$hyperVEnabled.HypervisorPresent) {
    Enable-WindowsOptionalFeature -Online -FeatureName VirtualMachinePlatform
}

# Enable Windows Subsystem for Linux if it is not already enabled.
if (!$wslEnabled.Enabled) {
    Enable-WindowsOptionalFeature -Online -FeatureName Microsoft-Windows-Subsystem-Linux
}

# Restart the computer if any of the features were changed.
if ($hyperVEnabled.HypervisorPresent -ne 
    $hyperVEnabled.HypervisorPresent -or 
    $wslEnabled.Enabled -ne $wslEnabled.Enabled) {
    Write-Host 'Restart Required. Press Enter to continue..'
    Read-Host
    Restart-Computer
}
